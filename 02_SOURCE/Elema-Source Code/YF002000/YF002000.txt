'01' TO W_DATE+6(2).

CALL FUNCTION 'FI_PERIOD_DETERMINE'
EXPORTING
I_BUDAT        = W_DATE
I_BUKRS        = P_BUKRS
IMPORTING
E_GJAHR        = W_GJAHR

EXCEPTIONS
FISCAL_YEAR    = 1

APPEND                       T_SKC1A.

END-OF-SELECTION.

PERFORM  EDT_HEADER.                 "見出し編集

LOOP  AT  T_BUMON.

PERFORM  READ_SKC1A.               "総勘定元帳→帳票出力用ﾜｰｸﾃｰﾌﾞﾙ

PERFORM  PRNT_LIST.                "損益計算書出力処理

*======================================================================*
*  ファイルの読込み                                                    *
*======================================================================*
FORM READ_FILES.

CLEAR  : T_BUMON,T_CYOHYO,T_SYUKEI.
REFRESH: T_BUMON,T_CYOHYO,T_SYUKEI.
*------------------------------------------部門ﾃｰﾌﾞﾙ
SELECT  *  FROM  TGSBT
WHERE  GSBER  BETWEEN  S_GSBERF  AND  S_GSBERT.

IF  SY-SUBRC  <>  0.
EXIT.
ENDIF.

T_BUMON-GSBER   =  TGSBT-GSBER.
T_BUMON-GTEXT   =  TGSBT-GTEXT.
*   T_BUMON-ZBBOOD  =  ZC250-ZBBOOD.

APPEND  T_BUMON.
ENDSELECT.
*------------------------------------------出力CTLﾃｰﾌﾞﾙ
SELECT  *  FROM  YFA03  INTO  T_CYOHYO.

IF  SY-SUBRC  =  0.
W_CTR  =  W_CTR  +  1.           "読込件数ｶｳﾝﾀ UP
ELSE.
EXIT.
ENDIF.

APPEND  T_CYOHYO.

IF  T_CYOHYO-ZBPRKBN  =  'B'.
W_IDX_BUNBO  =  W_CTR.
ENDIF.
ENDSELECT.
*------------------------------------------集計先決定ﾃｰﾌﾞﾙ
SELECT  *  FROM  YFA06  APPENDING  TABLE  T_ZF210.

IF  SY-SUBRC  NE  0.
EXIT.
ENDIF.
*------------------------------------------集計ﾃｰﾌﾞﾙ
DO  9  TIMES.
APPEND  T_SYUKEI.
ENDDO.
*------------------------------------------内部テーブルソート

SORT  T_BUMON  BY GSBER.             "部門ﾃｰﾌﾞﾙ  出力順にｿｰﾄ
SORT  T_ZF210  BY HKONT.
ENDFORM.

*======================================================================*
*  帳票出力用ﾜｰｸﾃｰﾌﾞﾙ  CLEAR                                           *
*======================================================================*
FORM INIT_T_CYOHYO.

LOOP  AT  T_CYOHYO.

T_CYOHYO-ZENZAN   =  0.
T_CYOHYO-KARI     =  0.
T_CYOHYO-KASI     =  0.
T_CYOHYO-TOUZAN   =  0.
T_CYOHYO-MATUZAN  =  0.
T_CYOHYO-HI       =  0.

MODIFY  T_CYOHYO.
ENDLOOP.

W_CTR_LINE  =  99.

ENDFORM.

*======================================================================*
*  総勘定元帳→帳票出力用ﾜｰｸﾃｰﾌﾞﾙ・当期利益計算用ﾜｰｸﾃｰﾌﾞﾙ              *
*======================================================================*
*FORM READ_SKC1A.
*
*  LOOP AT T_SKC1A  WHERE  GSBER  =  T_BUMON-GSBER.
*
*       AT END OF SAKNR.
*
*          SUM.
*         PERFORM  ZIMD0005.               "総勘定元帳の対象月データ取得
*
*          LOOP AT T_ZF210 WHERE HKONT    =   T_SKC1A-SAKNR
*                            AND ZBHKKEY <>  ''.
*
*            PERFORM  SUM_TUKI_DATA.        "対象月の金額集計
*
*          ENDLOOP.
*       ENDAT.
*  ENDLOOP.
*
*ENDFORM.
FORM READ_SKC1A.
*                                          "総勘定
LOOP AT T_SKC1A.
ZI05-ZENZAN  = 0.
ZI05-KARI    = 0.
ZI05-KASI    = 0.
ZI05-TOUZAN  = 0.
ZI05-MATUZAN = 0.
*現在の部門
IF T_SKC1A-GSBER  =  T_BUMON-GSBER.
PERFORM  ZIMD0005.               "総勘定元帳

LOOP AT T_ZF210 WHERE HKONT   =  T_SKC1A-SAKNR
AND ZBHKKEY <>  ''.
PERFORM  SUM_TUKI_DATA.        "対象月の金額
ENDLOOP.

ENDIF.
ENDLOOP.

ENDFORM.

*----------------------------------------------------------------------*
*  帳票出力用ﾜｰｸﾃｰﾌﾞﾙ金額集計                                          *
*----------------------------------------------------------------------*
FORM SUM_TUKI_DATA.

READ  TABLE T_CYOHYO  INDEX  T_ZF210-ZBHKSEQ.

T_CYOHYO-ZENZAN   =  T_CYOHYO-ZENZAN   +  ZI05-ZENZAN.
T_CYOHYO-KARI     =  T_CYOHYO-KARI     +  ZI05-KARI.
T_CYOHYO-KASI     =  T_CYOHYO-KASI     +  ZI05-KASI.
T_CYOHYO-TOUZAN   =  T_CYOHYO-TOUZAN   +  ZI05-TOUZAN.
T_CYOHYO-MATUZAN  =  T_CYOHYO-MATUZAN  +  ZI05-MATUZAN.

MODIFY      T_CYOHYO  INDEX  T_ZF210-ZBHKSEQ.

ENDFORM.

*======================================================================*
*  金額集計処理                                                        *
*======================================================================*
FORM SUMM_KINGAKU.
*- - - - - - - - - - - - - - - - - - - - - "ランクごとの金額集計
PERFORM  ZIMD0006.
*- - - - - - - - - - - - - - - - - - - - - "構成比計算
PERFORM  KEIS_KOSEIHI.

ENDFORM.

*----------------------------------------------------------------------*
*  構成比計算                                                          *
*----------------------------------------------------------------------*
FORM KEIS_KOSEIHI.
*                                          "構成比計算用分母取得
READ  TABLE T_CYOHYO  INDEX  W_IDX_BUNBO.

W_BUNBO  =  T_CYOHYO-MATUZAN.
*                                          "構成比計算
LOOP AT T_CYOHYO.

IF  W_BUNBO  <>  0.
T_CYOHYO-HI  =  T_CYOHYO-MATUZAN  /  W_BUNBO  *  100.
ENDIF.

MODIFY      T_CYOHYO  INDEX  SY-TABIX.
ENDLOOP.

ENDFORM.

*======================================================================*
*  部門別販売費及び一般管理費内訳  出力処理                            *
*======================================================================*
FORM PRNT_LIST.
*                                          "出力コントロール
LOOP AT T_CYOHYO  WHERE  ZBPRKBN  <> 'D'.

IF  W_CTR_LINE  >=  58.

NEW-PAGE.
PERFORM  OUT_HEADER.             "見出し出力
W_CTR_LINE  =  7.
ENDIF.

PERFORM  OUT_DETAIL.               "明細出力

W_CTR_LINE  =  W_CTR_LINE  +  1.   "明細行カウントＵＰ
ENDLOOP.

ENDFORM.

*----------------------------------------------------------------------*
*  見出し編集処理                                                      *
*----------------------------------------------------------------------*
FORM EDT_HEADER.

*                                          "月末日取得
CALL FUNCTION 'LAST_DAY_IN_PERIOD_GET'
EXPORTING
I_GJAHR        = W_NEN_F
I_PERIV        = T001-PERIV
I_POPER        = W_TUKI3

ENDIF.

ENDFORM.

*----------------------------------------------------------------------*
*  明細出力処理                                                        *
*----------------------------------------------------------------------*
FORM OUT_DETAIL.
*                                          "見出し編集
CASE    T_CYOHYO-ZBLSNM.
WHEN  '△                  '.
SKIP.
WHEN  '--------------------'.
ULINE /051(69).                  "単線出力
WHEN  '===================='.
WRITE: /051 '==============================',         "二重線出力
081 '==============================',
111 '========='.
WHEN OTHERS.
WRITE: /051  T_CYOHYO-ZBLSNM.

W_KINGAKU13 = T_CYOHYO-TOUZAN * 100.      "97/03/04
PERFORM YF01NUMC USING W_KINGAKU13 20 0.       "97/03/04
WRITE:  072  O_NUM USING EDIT MASK 'RR____________________'.

W_KINGAKU13 = T_CYOHYO-MATUZAN * 100.     "97/03/04
PERFORM YF01NUMC USING W_KINGAKU13 20 0.       "97/03/04
WRITE:  093  O_NUM USING EDIT MASK 'RR____________________'.

PERFORM  EDT_KOSEIHI.            "構成比
WRITE:  114  EDT USING EDIT MASK 'RR______'.

ENDCASE.

ENDFORM.

*----------------------------------------------------------------------*
*  構成比編集                                                          *
*----------------------------------------------------------------------*
FORM EDT_KOSEIHI.

IF T_CYOHYO-HI  <  0.                "負の値の場合
EDT-MINUS  =  '-'.                 "  -セット
W_HI     =  T_CYOHYO-HI  *  -1.    "　絶対値セット
EDT-HI     =         W_HI.
ELSE.                                "正の値の場合
EDT-MINUS  =  ''.
EDT-HI     =  T_CYOHYO-HI.
ENDIF.

CONDENSE  EDT  NO-GAPS.

ENDFORM.

*======================================================================*

*======================================================================*
FORM INIT_T_CYOHYO.

LOOP  AT  T_CYOHYO.

T_CYOHYO-ZENZAN   =  0.
T_CYOHYO-KARI     =  0.
T_CYOHYO-KASI     =  0.

*======================================================================*
*  総勘定元帳→帳票出力用ﾜｰｸﾃｰﾌﾞﾙ・当期利益計算用ﾜｰｸﾃｰﾌﾞﾙ              *
*======================================================================*
*FORM READ_SKC1A.
*
*  LOOP AT T_SKC1A  WHERE  GSBER  =  T_BUMON-GSBER.
*
*       AT END OF SAKNR.
*
*          SUM.
*         PERFORM  ZIMD0005.               "総勘定元帳の対象月データ取得
*
*          LOOP AT T_ZF210 WHERE HKONT    =   T_SKC1A-SAKNR
*                            AND ZBHKKEY <>  ''.
*
*            PERFORM  SUM_TUKI_DATA.        "対象月の金額集計
*
*          ENDLOOP.
*       ENDAT.
*  ENDLOOP.
*
*ENDFORM.
FORM READ_SKC1A.
*                                          "総勘定
LOOP AT T_SKC1A.
ZI05-ZENZAN  = 0.
ZI05-KARI    = 0.
ZI05-KASI    = 0.
ZI05-TOUZAN  = 0.
ZI05-MATUZAN = 0.
*現在の部門
IF T_SKC1A-GSBER  =  T_BUMON-GSBER.
PERFORM  ZIMD0005.               "総勘定元帳

LOOP AT T_ZF210 WHERE HKONT   =  T_SKC1A-SAKNR
AND ZBHKKEY <>  ''.
PERFORM  SUM_TUKI_DATA.        "対象月の金額
ENDLOOP.

ENDIF.
ENDLOOP.

ENDFORM.

*----------------------------------------------------------------------*
*  帳票出力用ﾜｰｸﾃｰﾌﾞﾙ金額集計                                          *

FORM SUM_TUKI_DATA.

READ  TABLE T_CYOHYO  INDEX  T_ZF210-ZBHKSEQ.

T_CYOHYO-ZENZAN   =  T_CYOHYO-ZENZAN   +  ZI05-ZENZAN.
T_CYOHYO-KARI     =  T_CYOHYO-KARI     +  ZI05-KARI.
T_CYOHYO-KASI     =  T_CYOHYO-KASI     +  ZI05-KASI.
T_CYOHYO-TOUZAN   =  T_CYOHYO-TOUZAN   +  ZI05-TOUZAN.
T_CYOHYO-MATUZAN  =  T_CYOHYO-MATUZAN  +  ZI05-MATUZAN.

ENDFORM.

*----------------------------------------------------------------------*
*  構成比計算                                                          *
*----------------------------------------------------------------------*
FORM KEIS_KOSEIHI.
*                                          "構成比計算用分母取得
READ  TABLE T_CYOHYO  INDEX  W_IDX_BUNBO.

W_BUNBO  =  T_CYOHYO-MATUZAN.
*                                          "構成比計算
LOOP AT T_CYOHYO.

T_CYOHYO-HI  =  T_CYOHYO-MATUZAN  /  W_BUNBO  *  100.
ENDIF.

MODIFY      T_CYOHYO  INDEX  SY-TABIX.
ENDLOOP.

ENDFORM.

*----------------------------------------------------------------------*
*  構成比編集                                                          *
*----------------------------------------------------------------------*
FORM EDT_KOSEIHI.

IF T_CYOHYO-HI  <  0.                "負の値の場合
EDT-MINUS  =  '-'.                 "  -セット
W_HI     =  T_CYOHYO-HI  *  -1.    "　絶対値セット
EDT-HI     =         W_HI.
ELSE.                                "正の値の場合
EDT-MINUS  =  ''.
EDT-HI     =  T_CYOHYO-HI.
ENDIF.

CONDENSE  EDT  NO-GAPS.

ENDFORM.
