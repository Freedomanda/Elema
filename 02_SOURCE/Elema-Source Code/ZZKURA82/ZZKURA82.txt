REPORT zzkura82 .

TABLES lfm1.  "仕入先マスタ購買組織データ
TABLES t001w. "プラント/支店

DATA i_t001w LIKE STANDARD TABLE OF t001w WITH HEADER LINE.
DATA w_lifnr LIKE lfm1-lifnr.
DATA i_yk250 LIKE STANDARD TABLE OF yk250 WITH HEADER LINE.
DATA w_cnt1 LIKE sy-dbcnt. "仕入先件数
DATA w_cnt2 LIKE sy-dbcnt. "エラー件数
DATA w_cnt3 LIKE sy-dbcnt. "登録件数

SELECT-OPTIONS s_werks FOR t001w-werks.
SELECT-OPTIONS s_lifnr FOR lfm1-lifnr.
PARAMETERS p_ekorg LIKE lfm1-ekorg DEFAULT '1000'.
PARAMETERS p_bukrs LIKE yk250-bukrs DEFAULT 'C001'.
PARAMETERS p_werksp LIKE yk250-werksp DEFAULT 'PS00'.
PARAMETERS p_test AS CHECKBOX DEFAULT 'X'.

*----------------------------------------------------------------------*
INITIALIZATION.
s_werks-sign = 'I'.
s_werks-option = 'CP'.
s_werks-low = 'PS*'.
APPEND s_werks.

START-OF-SELECTION.

SELECT        *
FROM   t001w
INTO   TABLE i_t001w
WHERE  werks IN s_werks.
WRITE: / 'プラント件数', sy-dbcnt.
SELECT lifnr
INTO   w_lifnr
FROM   lfm1
WHERE  lifnr IN s_lifnr
AND    ekorg  = p_ekorg.
LOOP AT i_t001w.
CLEAR i_yk250.
i_yk250-bukrs = p_bukrs.
i_yk250-lifnr = w_lifnr.
i_yk250-werkst = i_t001w-werks.
i_yk250-werksp = p_werksp.
i_yk250-insdat = sy-datum.
i_yk250-instim = sy-uzeit.
i_yk250-insusr = sy-uname.
APPEND i_yk250.
ENDLOOP.
ADD 1 TO w_cnt1.
ENDSELECT.
WRITE: / '仕入先件数　', w_cnt1.
IF p_test <> 'X'.
LOOP AT i_yk250.
INSERT yk250 FROM i_yk250.
IF sy-subrc <> 0.
ADD 1 TO w_cnt2.
ELSE.
ADD 1 TO w_cnt3.
ENDIF.
ENDLOOP.
ENDIF.
WRITE: / 'エラー件数　', w_cnt2.
WRITE: / '登録件数　　', w_cnt3.
