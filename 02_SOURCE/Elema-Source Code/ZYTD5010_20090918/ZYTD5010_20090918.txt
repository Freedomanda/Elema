FORM DATA_GET .

DATA : W_MBEW TYPE MBEW .

CLEAR TAB.
SELECT WERKS EKGRP MATNR MTVFP
* Add 2007/10/01 --->
HERKL
STAWN
* Add 2007/10/01 <---
FROM MARC
INTO CORRESPONDING FIELDS OF TABLE TAB
WHERE WERKS EQ I_WERKS
AND  MATNR IN S_MATNR.
SELECT *
INTO TABLE T_MARA
FROM MARA FOR ALL ENTRIES IN TAB
WHERE MATNR EQ TAB-MATNR.

LOOP AT TAB .
* Add 2008.03.14 >>>
IF TAB-HERKL IS INITIAL .
MOVE 'JP' TO TAB-HERKL .
ENDIF .
CLEAR : W_MBEW .
SELECT LGPBE
FROM MARD
INTO CORRESPONDING FIELDS OF TAB
WHERE WERKS EQ I_WERKS
AND  MATNR EQ TAB-MATNR .
EXIT .
ENDSELECT .
IF SY-SUBRC NE 0.
MOVE :    '1'     TO TAB-LGPBE .
ENDIF.
SELECT VERPR PEINH
FROM MBEW
INTO CORRESPONDING FIELDS OF W_MBEW
WHERE BWKEY EQ I_WERKS
AND  MATNR EQ TAB-MATNR .
EXIT .
ENDSELECT .
IF SY-SUBRC EQ 0.
MOVE :  W_MBEW-VERPR TO TAB-VERPR ,
W_MBEW-PEINH TO TAB-PEINH .
MULTIPLY : TAB-VERPR By 100 .
ELSE .
MOVE :  1      TO TAB-VERPR .
ENDIF.
* Add 2008.03.14 <<<
READ TABLE T_MARA WITH KEY MATNR = TAB-MATNR .
MOVE-CORRESPONDING T_MARA TO TAB .

MOVE : P_WERKS TO TAB-WERKS ,
*           P_EKGRP To TAB-EKGRP ,
P_WERKS+1(3) TO TAB-EKGRP ,
* Mod 2008.03.14 >>>
*           1       TO TAB-VERPR ,
*           '1'     TO TAB-LGPBE ,
* Mod 2008.03.14 <<<
'X'     TO TAB-KAUTB ,
'/'     TO TAB-MVGR1 ,
'/'     TO TAB-MVGR2 ,
'/'     TO TAB-MVGR3 ,
'/'     TO TAB-MVGR4 ,
'/'     TO TAB-PRODH ,
'/'     TO TAB-MTPOS ,
'/'     TO TAB-VPRSV ,
'/'     TO TAB-PEINH ,
'/'     TO TAB-MTVFP ,
'/'     TO TAB-VERSG ,
'/'     TO TAB-BKLAS ,
'MWST'  TO TAB-TATY1 .

ADD 1 TO TOTAL .
CLEAR : T_MARA .
MODIFY TAB .
ENDLOOP .
ENDFORM .
