REPORT ZTEST_20090527 .

TYPES:BEGIN OF T_MARA,
MATNR TYPE MARA-MATNR,
END OF T_MARA.
DATA:IT_MARA TYPE TABLE OF T_MARA,
WA_MARA TYPE T_MARA,
V_MATNR TYPE YS0011-MATNR.

SELECT MATNR
INTO TABLE IT_MARA
FROM MARA.

LOOP AT IT_MARA INTO WA_MARA.
SELECT MATNR
INTO V_MATNR
FROM YS0011
UP TO 1 ROWS
WHERE MATNR = WA_MARA-MATNR.
ENDSELECT.
IF SY-SUBRC <> 0.
WRITE : / WA_MARA-MATNR.
ENDIF.
ENDLOOP.
