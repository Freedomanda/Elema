*&      Form  GET_MKPF
*&---------------------------------------------------------------------*
*       入出庫伝票情報：ヘッダ
*----------------------------------------------------------------------*
FORM GET_MKPF.

REFRESH IT_MKPF.
SELECT MBLNR         "入出庫伝票番号
MJAHR         "入出庫伝票年度
BLDAT         "伝票日付
BUDAT         "転記日付
CPUDT         "会計伝票登録日
CPUTM         "登録時刻
AEDAT         "変更日
USNAM         "ユーザ名
INTO TABLE IT_MKPF
FROM MKPF
WHERE BUDAT IN S_BUDAT
AND MBLNR IN S_MBLNR
AND MJAHR IN S_MJAHR
AND BLDAT IN S_BLDAT
AND CPUDT IN S_CPUDT
AND CPUTM IN S_CPUTM
AND AEDAT IN S_AEDAT
AND USNAM IN S_USNAM.

IF SY-SUBRC <> 0.
*   処理対象データがありません
MESSAGE S204.
STOP.
ENDIF.

ENDFORM.                    " GET_MKPF
*&---------------------------------------------------------------------*
*&      Form  GET_MSEG
*&---------------------------------------------------------------------*
*       入出庫伝票情報：明細
*----------------------------------------------------------------------*
FORM GET_MSEG.

REFRESH IT_MSEG.
SELECT MBLNR         "入出庫伝票番号
MJAHR         "入出庫伝票年度
ZEILE         "入出庫伝票明細
BWART         "移動タイプ
MATNR         "品目コード
WERKS         "プラント
LIFNR         "仕入先
KUNNR         "得意先
KDAUF         "受注伝票番号
DMBTR         "国内通貨額
ERFMG         "入力単位での数量
EBELN         "購買発注伝票番号
INTO TABLE IT_MSEG
FROM MSEG
FOR ALL ENTRIES IN IT_MKPF
WHERE MBLNR =  IT_MKPF-MBLNR
AND MJAHR =  IT_MKPF-MJAHR
AND MATNR IN S_MATNR
AND WERKS IN S_WERKS
AND BWART IN S_BWART
AND LIFNR IN S_LIFNR
AND KUNNR IN S_KUNNR
AND EBELN IN S_EBELN.

IF SY-SUBRC <> 0.
*   処理対象データがありません
MESSAGE S204.
STOP.
ENDIF.

ENDFORM.                    " GET_MSEG
*&---------------------------------------------------------------------*
*&      Form  GET_EKKO
*&---------------------------------------------------------------------*
*       購買伝票情報
*----------------------------------------------------------------------*
FORM GET_EKKO.

REFRESH IT_EKKO.
SELECT EKKO~EBELN         "購買伝票番号
EKKO~EKGRP         "購買グループ
EKKO~BEDAT         "発注伝票日付
EKPO~NETPR         "正味価格
EKPO~PEINH         "価格単位
INTO TABLE IT_EKKO
FROM EKKO
INNER JOIN EKPO
ON EKKO~EBELN = EKPO~EBELN
AND EKKO~LPONR = EKPO~EBELP
FOR ALL ENTRIES IN IT_MSEG
WHERE EKKO~EBELN =  IT_MSEG-EBELN.

ENDFORM.                    " GET_EKKO
*&---------------------------------------------------------------------*
*&      Form  GET_LFA1
*&---------------------------------------------------------------------*
*       通貨・仕入先名称
*----------------------------------------------------------------------*
FORM GET_LFA1.

DATA : LT_MSEG TYPE STANDARD TABLE OF T_MSEG.

* 仕入先の重複を削除
LT_MSEG = IT_MSEG.
SORT LT_MSEG BY LIFNR.
DELETE ADJACENT DUPLICATES FROM LT_MSEG COMPARING LIFNR.

REFRESH IT_LFA1.
SELECT LFM1~LIFNR
LFM1~WAERS
LFA1~NAME1
INTO TABLE IT_LFA1
FROM LFM1
INNER JOIN LFA1
ON LFM1~LIFNR = LFA1~LIFNR
FOR ALL ENTRIES IN LT_MSEG
WHERE LFM1~LIFNR = LT_MSEG-LIFNR
AND LFM1~EKORG = CNS_EKORG.

ENDFORM.                    " GET_LFA1
*&---------------------------------------------------------------------*
*&      Form  GET_MAKT
*&---------------------------------------------------------------------*
*       品目テキスト
*----------------------------------------------------------------------*
FORM GET_MAKT.

DATA : LT_MSEG TYPE STANDARD TABLE OF T_MSEG.

* 品目コードの重複を削除
LT_MSEG = IT_MSEG.
SORT LT_MSEG BY MATNR.
DELETE ADJACENT DUPLICATES FROM LT_MSEG COMPARING MATNR.

REFRESH IT_MAKT.
SELECT MATNR         "品目コード
MAKTX         "品目テキスト
INTO TABLE IT_MAKT
FROM MAKT
FOR ALL ENTRIES IN LT_MSEG
WHERE MATNR = LT_MSEG-MATNR
AND SPRAS = CNS_SPRAS_J.

ENDFORM.                    " GET_MAKT
*&---------------------------------------------------------------------*
*&      Form  GET_MARD
*&---------------------------------------------------------------------*
*       棚番
*----------------------------------------------------------------------*
FORM GET_MARD.

DATA : LT_MSEG TYPE STANDARD TABLE OF T_MSEG.

* 品目・プラントの重複を削除
LT_MSEG = IT_MSEG.
SORT LT_MSEG BY MATNR WERKS.
DELETE ADJACENT DUPLICATES FROM LT_MSEG COMPARING MATNR WERKS.

REFRESH IT_MARD.
SELECT MATNR         "品目コード
WERKS         "プラント
LGPBE         "棚番
INTO TABLE IT_MARD
FROM MARD
FOR ALL ENTRIES IN LT_MSEG
WHERE MATNR = LT_MSEG-MATNR
AND WERKS = LT_MSEG-WERKS.

ENDFORM.                    " GET_MARD

*&      Form  EDIT_DATA
*&---------------------------------------------------------------------*
*       編集
*----------------------------------------------------------------------*
FORM EDIT_DATA.

DATA : LV_KIN_C(20) TYPE C,
LV_KIN_P     TYPE P DECIMALS 2.

REFRESH IT_ALV.
LOOP AT IT_MSEG INTO WA_MSEG.
CLEAR: WA_ALV.

*   入出庫伝票情報：ヘッダ
CLEAR WA_MKPF.
READ TABLE IT_MKPF INTO WA_MKPF
WITH TABLE KEY MBLNR = WA_MSEG-MBLNR
MJAHR = WA_MSEG-MJAHR.
CHECK SY-SUBRC = 0.

*   購買伝票情報
CLEAR WA_EKKO.
READ TABLE IT_EKKO INTO WA_EKKO
WITH TABLE KEY EBELN = WA_MSEG-EBELN.
*   購買伝票が存在する場合、
*   購買グループが選択画面指定値に該当しないものは除外
IF SY-SUBRC      = 0.
IF WA_EKKO-EKGRP IN S_EKGRP.
ELSE.
*       処理対象外
CONTINUE.
ENDIF.
ENDIF.

*   通貨コード・仕入先名
CLEAR WA_LFA1.
READ TABLE IT_LFA1 INTO WA_LFA1
WITH TABLE KEY LIFNR = WA_MSEG-LIFNR.
*   品目テキスト
CLEAR WA_MAKT.
READ TABLE IT_MAKT INTO WA_MAKT
WITH TABLE KEY MATNR = WA_MSEG-MATNR.
*   棚番
CLEAR WA_MARD.
READ TABLE IT_MARD INTO WA_MARD
WITH TABLE KEY MATNR = WA_MSEG-MATNR
WERKS = WA_MSEG-WERKS.

WA_ALV-WERKS   = WA_MSEG-WERKS.                   "プラント
WA_ALV-MATNR   = WA_MSEG-MATNR.                   "品目コード
WA_ALV-MAKTX   = WA_MAKT-MAKTX.                   "品名
WA_ALV-EKGRP   = WA_EKKO-EKGRP.                   "購買グループ
WA_ALV-BWART   = WA_MSEG-BWART.                   "移動タイプ
WA_ALV-BUDAT   = WA_MKPF-BUDAT.                   "転記日付
WA_ALV-MBLNR   = WA_MKPF-MBLNR.                   "入出庫伝票番号
WA_ALV-ERFMG   = WA_MSEG-ERFMG.                   "数量
WA_ALV-WAERS   = WA_LFA1-WAERS.                   "通貨コード
CLEAR LV_KIN_C.
WRITE WA_EKKO-NETPR TO LV_KIN_C
CURRENCY WA_ALV-WAERS NO-GROUPING.
WA_ALV-NETPR   = LV_KIN_C.                        "正味価格
WA_ALV-PEINH   = WA_EKKO-PEINH.                   "価格単位
WA_ALV-DMBTR   = WA_MSEG-DMBTR * 100.             "国内通貨額
WA_ALV-EBELN   = WA_EKKO-EBELN.                   "購買伝票番号
WA_ALV-LIFNR   = WA_MSEG-LIFNR.                   "仕入先勘定コード
WA_ALV-NAME1   = WA_LFA1-NAME1.                   "仕入先名
WA_ALV-KDAUF   = WA_MSEG-KDAUF.                   "受注伝票番号
WA_ALV-MJAHR   = WA_MKPF-MJAHR.                   "入出庫伝票年度
WA_ALV-KUNNR   = WA_MSEG-KUNNR.                   "得意先コード
WA_ALV-BLDAT   = WA_MKPF-BLDAT.                   "伝票日付
WA_ALV-CPUDT   = WA_MKPF-CPUDT.                   "伝票登録日
WA_ALV-CPUTM   = WA_MKPF-CPUTM.                   "登録時間
WA_ALV-AEDAT   = WA_MKPF-AEDAT.                   "変更日
WA_ALV-USNAM   = WA_MKPF-USNAM.                   "ユーザ名
WA_ALV-LGPBE   = WA_MARD-LGPBE.                   "棚番
WA_ALV-BEDAT   = WA_EKKO-BEDAT.                   "発注伝票日付　
PERFORM GET_UKURS USING    WA_ALV-WAERS
WA_ALV-BUDAT
CHANGING WA_ALV-UKURS.          "換算レート
CLEAR : LV_KIN_C,LV_KIN_P.
IF WA_EKKO-PEINH = 0.
LV_KIN_P = 0.
ELSE.
LV_KIN_P       = WA_EKKO-NETPR / WA_EKKO-PEINH
* WA_MSEG-ERFMG.
ENDIF.
WRITE LV_KIN_P TO LV_KIN_C CURRENCY WA_ALV-WAERS
NO-GROUPING.
WA_ALV-INGAKU  = LV_KIN_C.                        "入庫金額
WA_ALV-INGAKU2 = WA_ALV-INGAKU * WA_ALV-UKURS.    "入庫金額円建
APPEND WA_ALV TO IT_ALV.
ENDLOOP.

* 出力対象データ有無チェック
IF IT_ALV IS INITIAL.
*   処理対象データがありません
MESSAGE S204.
STOP.
ENDIF.
FREE: IT_MKPF,IT_MSEG,IT_EKKO,IT_LFA1,IT_MAKT,IT_MARD.

ENDFORM.                    " EDIT_DATA

CASE WA_EVENT-NAME.
WHEN 'TOP_OF_PAGE'.
WA_EVENT-FORM = 'OUT_HEADER'.
MODIFY IT_EVENT FROM WA_EVENT.

WHEN 'CALLER_EXIT'.
WA_EVENT-FORM = 'CALLER_EXIT'.
MODIFY IT_EVENT FROM WA_EVENT.

WHEN OTHERS.
ENDCASE.
ENDLOOP.
