*    WORK = STAB-KUNNR.
*    IF WORK CO '0123456789'.
*      W_KUNNR = '0000000000'.
*      W_KUNNR = WORK.
*    ELSE.

EXPORTING
INPUT         = WORK
IMPORTING
OUTPUT        = W_KUNNR.
***
STAB-KUNNR = W_KUNNR.
***2002.05.23 EDIT >>>

*    SELECT SINGLE * FROM KNA1 WHERE KUNNR = W_KUNNR.

*    IF SY-SUBRC = 0.
*      MESSAGE I402 WITH '得意先マスタに存在します.得意=' STAB-KUNNR.
