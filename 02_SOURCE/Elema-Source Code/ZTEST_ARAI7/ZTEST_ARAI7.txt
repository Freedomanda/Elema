REPORT ZTEST_ARAI7 .

DATA : ZNEN002T LIKE TABLE OF ZNEN002.
DATA : ZNEN002W LIKE ZNEN002.
DATA : MARAW LIKE MARA.
DATA : EKPOW LIKE EKPO.

SELECT * FROM ZNEN002
INTO TABLE ZNEN002T
WHERE ZZNO NE SPACE.

LOOP AT ZNEN002T INTO ZNEN002W.

SELECT SINGLE * FROM MARA
INTO MARAW
WHERE MATNR = ZNEN002W-MATNR.

SELECT SINGLE * FROM EKPO
INTO EKPOW
WHERE EBELN = ZNEN002W-ZZNO
AND EBELP = ZNEN002W-ZZPOS.

IF SY-SUBRC NE 0.
CONTINUE.
ENDIF.

IF EKPOW-MEINS NE MARAW-MEINS.
WRITE : / ZNEN002W-ZZNO.
ENDIF.
ENDLOOP.
