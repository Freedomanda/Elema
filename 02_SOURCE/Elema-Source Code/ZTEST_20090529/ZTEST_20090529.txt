REPORT ZTEST_20090529 .

TYPES:BEGIN OF T_MARA,
MATNR TYPE MARA-MATNR,
BWKEY TYPE MBEW-BWKEY,
VERPR TYPE MBEW-VERPR,
END OF T_MARA.

DATA:IT_MARA TYPE TABLE OF T_MARA,
WA_MARA TYPE T_MARA,
IT_MBEW TYPE TABLE OF T_MARA,
WA_MBEW TYPE T_MARA.

SELECT MATNR
INTO TABLE IT_MARA
FROM MARA.

LOOP AT IT_MARA INTO WA_MARA.
SELECT MATNR BWKEY VERPR
INTO TABLE IT_MBEW
FROM MBEW
WHERE MATNR = WA_MARA-MATNR.

LOOP AT IT_MBEW INTO WA_MBEW.
IF WA_MBEW-VERPR IS INITIAL.
WRITE : / WA_MBEW-MATNR,WA_MBEW-BWKEY.
ENDIF.
ENDLOOP.
ENDLOOP.
