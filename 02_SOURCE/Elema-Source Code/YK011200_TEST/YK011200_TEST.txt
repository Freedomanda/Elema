*&      Form  GET_W_SPART

*       製品部門変換 2003/05/013 ADD
*----------------------------------------------------------------------*
FORM GET_INIT_CHK.
CLEAR W_SPART.
SELECT SINGLE SPAKU FROM TVTA INTO  W_SPART
WHERE VKORG = P_VKORG
AND VTWEG = P_VTWEG
AND SPART = P_SPART.
IF SY-SUBRC <> 0.
W_SPART = P_SPART.
ENDIF.
CLEAR YK150.
SELECT SINGLE B_VKBUR_INIT B_VKGRP_INIT
INTO (YK150-B_VKBUR_INIT,YK150-B_VKGRP_INIT) FROM YK150
WHERE BUKRS = P_BUKRS
AND KUNNR = S_KUNNR-LOW
AND VKORG = P_VKORG
AND VTWEG = P_VTWEG
AND SPART = P_SPART.
IF SY-SUBRC <> 0.
SELECT  B_VKBUR_INIT B_VKGRP_INIT
INTO (YK150-B_VKBUR_INIT,YK150-B_VKGRP_INIT) FROM YK150
UP TO 1 ROWS
WHERE KUNNR = 'DEFAULT'.
ENDSELECT.
ENDIF.

ENDFORM.                    " GET_W_SPART
