*======================================================================*
*  金額集計処理                                                        *
*======================================================================*
FORM SUMM_KINGAKU.
*- - - - - - - - - - - - - - - - - - - - - "当期利益(損失)の金額設定
READ  TABLE T_CYOHYO  INDEX  W_IDX_RIEKI.

T_CYOHYO-ZENZAN   =  T_RIEKI-ZENZAN   * -1.   "後で符号反転されるので
T_CYOHYO-TOUZAN   =  T_RIEKI-TOUZAN   * -1.   "あらかじめ逆転しておく

IF  T_RIEKI-TOUZAN  <  0.
T_CYOHYO-KARI  =  T_RIEKI-TOUZAN  *  -1.
ELSE.
T_CYOHYO-KASI  =  T_RIEKI-TOUZAN.
ENDIF.

MODIFY      T_CYOHYO  INDEX  W_IDX_RIEKI.

*- - - - - - - - - - - - - - - - - - - - - "構成比計算
PERFORM  KEIS_KOSEIHI.

ENDFORM.

*----------------------------------------------------------------------*
*  構成比計算                                                          *
*----------------------------------------------------------------------*
FORM KEIS_KOSEIHI.
*                                          "構成比計算用分母取得
READ  TABLE T_CYOHYO  INDEX  W_IDX_BUNBO.

W_BUNBO  =  T_CYOHYO-MATUZAN.
*                                          "構成比計算
LOOP AT T_CYOHYO.

IF  W_BUNBO  <>  0.
IF  T_CYOHYO-ZBDCKBN  =  'D'.
W_FUGO  =  -1.
ELSE.
W_FUGO  =   1.
ENDIF.

T_CYOHYO-HI = T_CYOHYO-MATUZAN / W_BUNBO * 100 * W_FUGO.
ENDIF.

MODIFY      T_CYOHYO  INDEX  SY-TABIX.
ENDLOOP.

ENDFORM.

*----------------------------------------------------------------------*

FORM EDT_KOSEIHI.

IF T_CYOHYO-HI  <  0.                "負の値の場合

*----------------------------------------------------------------------*
*  構成比計算                                                          *
*----------------------------------------------------------------------*
FORM KEIS_KOSEIHI.
*                                          "構成比計算用分母取得
READ  TABLE T_CYOHYO  INDEX  W_IDX_BUNBO.

W_BUNBO  =  T_CYOHYO-MATUZAN.
*                                          "構成比計算
LOOP AT T_CYOHYO.

IF  W_BUNBO  <>  0.
IF  T_CYOHYO-ZBDCKBN  =  'D'.
W_FUGO  =  -1.
ELSE.
W_FUGO  =   1.
ENDIF.

T_CYOHYO-HI = T_CYOHYO-MATUZAN / W_BUNBO * 100 * W_FUGO.
ENDIF.

MODIFY      T_CYOHYO  INDEX  SY-TABIX.
ENDLOOP.

ENDFORM.
